方案2：设置Xauthority文件
# 复制kali用户的Xauthority到root
cd /home/kali/Desktop
cp /home/kali/.Xauthority /root/

# 设置正确的权限
chown root:root /root/.Xauthority

# 设置DISPLAY变量
export DISPLAY=:0
export XAUTHORITY=/root/.Xauthority

./5ire-0.13.2-x86_64.AppImage --no-sandbox
